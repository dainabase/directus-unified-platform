# Text Animations

Animated text effects

## Import

```tsx
import { TextAnimation } from '@dainabase/ui/text-animations';
```

## Basic Usage

```tsx
import { TextAnimation } from '@dainabase/ui/text-animations';

export default function TextAnimationExample() {
  return (
    <TextAnimation 
      text="Animated Text"
      animation="fade-in"
      duration={1000}
    />
  );
}
```
