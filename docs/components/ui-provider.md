# UI Provider

UI context provider

## Import

```tsx
import { UIProvider } from '@dainabase/ui/ui-provider';
```

## Basic Usage

```tsx
import { UIProvider } from '@dainabase/ui/ui-provider';

export default function App() {
  return (
    <UIProvider theme="dark">
      <YourApp />
    </UIProvider>
  );
}
```
