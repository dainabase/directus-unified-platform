# Toggle

Toggle button component

## Import

```tsx
import { Toggle } from '@dainabase/ui/toggle';
```

## Basic Usage

```tsx
import { Toggle } from '@dainabase/ui/toggle';

export default function ToggleExample() {
  return (
    <Toggle aria-label="Toggle italic">
      <Icon icon={Italic} />
    </Toggle>
  );
}
```
